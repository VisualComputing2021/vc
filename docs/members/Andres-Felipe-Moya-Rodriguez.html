linkIlusión

En 1904, Mary Everest Boole (matemática inglesa) inventó las Curve stitching (curva de costura), una forma de arte donde se "crean" líneas curvas a través de líneas rectas. Este fenómeno de crear una aparente curvatura ocurre gracias al hecho que una curva se puede aproximar mediante el trazo de una cantidad finita de tangentes a la misma.

link3Doge

¡Puede interactuar con el entorno 3D!

  • Mueva la cámara con el mouse
  • Puede moverse con las flechas de su tecládo o las teclas awsd
  • Puede saltar con la barra de espacio
  • ¡Oprima el botón Rotar doge y observe lo que ocurre con la imagen!

Si no puede interactuar con el entorno, haga click dentro de él

1linklet angle = 0;

2linklet img;

3linklet camX;

4linklet camY;

5linklet camZ;

6linklet camPosX = 0;

7linklet camPosY = 0;

8linklet jump = false;

9linklet jumpCount = 0;

10linklet button;

11linklet rotate = false;

12linkfunction preload(){

13link  img = loadImage("doge.jpg");

15linkfunction setup() {

16link  // crea un lienzo de 400 pixeles de ancho, 300 de alto configurado para mostrar gráficos en 3D

17link  createCanvas(400, 300, WEBGL);

18link  camZ = (height/2)/tan(PI/6);

19link  button = createButton('Rotar doge');

20link  button.position(0, 0);

21link  button.mousePressed(changeBG);

22link}

23linkfunction draw() {

24link  //Fondo del canvas color gris

25link  background(175);

26link  //valores de x, y que varían mientras se mueve el mouse

27link  camX = map(mouseX, 0, width, -200, 200);

28link  camY = map(mouseY, 0, height, -200, 200);

29link  planeSide = 200;

30link  let fieldOfView = PI / 3.0;

31link  //Controles del juego para mover cámara

32link  if (keyIsDown(LEFT_ARROW) || keyIsDown(65)) {

33link    camPosX-=3;

34link  }else if(keyIsDown(RIGHT_ARROW) || keyIsDown(68) ) {

35link    camPosX+=3;

36link  }else if(keyIsDown(UP_ARROW)|| keyIsDown(87)){

37link    camZ-=3;

38link  }else if(keyIsDown(DOWN_ARROW)|| keyIsDown(83)){

39link    camZ+=3;

40link  }

41link  //Controla la función de saltar

42link  if(jump){

43link    if(jumpCount < 30){

44link      camPosY--;

45link    }else{

46link      camPosY++;

47link    }

48link    if(camPosY == 0){

49link      jump = false;

50link      jumpCount = 0;

51link    }

52link    jumpCount++;

53link  }

54link  /*

55link  posición de la camara: (camPosX,0,camZ)

56link  dirección donde la cámara apunta (camX + camPosX,camY,0)

57link  */

58link  camera(camPosX,camPosY,camZ,camX + camPosX,camY + camPosY,0,0,1,0);

59link  //Modifica el campo de visión según mueve el mouse, aspect ratio (w/h), y planos que truncan el fov (height/2)/(10*tan(PI/6)), (10*height/2)/(tan(PI/6)))

60link   perspective(fieldOfView, width / height, camZ/10, 10* camZ);

61link  //no dibuja los bordes de la figura ni sus primitivas

62link  noStroke();

63link  if(rotate){

64link    //hace rotar las figuras

65link    rotateX(angle);

66link    rotateY(angle/2);

67link    rotateZ(angle*2);

68link  }

69link  //Normaliza las coordenadas de la imagen de textura para mapearlas en la figura

70link  //carga la imagen como skin de la figura

71link  texture(img);

72link  textureMode(NORMAL); 

73link  beginShape();

74link  vertex(-planeSide/2, -planeSide/2, 0, 0); // esquina inferior izquierda

75link  vertex(planeSide/2, -planeSide/2, 1, 0); // esquina inferior derecha

76link  vertex(planeSide/2, planeSide/2, 1, 1); // esquina superior derecha

77link  vertex(-planeSide/2, planeSide/2, 0, 1); // esquina superior izquierda

78link  endShape(CLOSE);

79link  angle += 0.007 ;

80link}

84link    jump = true;

85link  }

86link}
