linkImagen

Mediante el canvas 3d que proporciona WEBGL se genera una figura rectangular a partir de cuatro vértices. Esta figura va a servir como "marco" para contener la imagen que se quiere mostrar. La imagen se define como una textura que se va aplicar al rectángulo.

Al igual que el anterior metodo de RGB aqui usamos la variable varying vTexCoord para poder acceder a las coordenadas, y posteriormente el color de cada texel de la imagen y luego poder operar sobre él haciendo un promedio ponderado en el primer de los casos con la fórmula \[ r * 0.299 + g * 0.587 + b * 0.0114 \] de los valores de cada canal rgb y asignándoselo al fragmento (pixel) correspondiente y luego en el LUMA normalizado usando \[ r' * 0.299 + g' * 0.587 + b' * 0.0114 \] dónde \[ r' = 255 * ( \frac{r}{255}) ^ {1/2.2} \] \[ g' = 255 * ( \frac{g}{255}) ^ {1/2.2} \] \[ b' = 255 * ( \frac{b}{255}) ^ {1/2.2} \].

1link// texture.frag 

2linkprecision mediump float;

3linkuniform sampler2D texture;

4link// interpolated color (same name as in vertex shader)

5linkvarying vec4 vVertexColor;

6linkvec4 grayTextureColor;

7linkfloat gray;

8link// interpolated texcoord (same name as in vertex shader)

9linkvarying vec2 vTexCoord;

10linkvoid main() { 

11link  grayTextureColor = texture2D(texture, vTexCoord);

12link  gray = (grayTextureColor.r*0.299 + grayTextureColor.g*0.587 + grayTextureColor.b*0.0114);

13link  grayTextureColor.r = gray;

14link  grayTextureColor.g = gray;

15link  grayTextureColor.b = gray;

16link  grayTextureColor.a = 1.0;

17link  gl_FragColor = grayTextureColor * vVertexColor;  

18link}

1link// texture.frag 

2linkprecision mediump float;

3linkuniform sampler2D texture;

4link// interpolated color (same name as in vertex shader)

5linkvarying vec4 vVertexColor;

6linkvec4 grayTextureColor;

7linkfloat gray;

8linkfloat red;

9linkfloat gr;

10linkfloat blue;

11linkfloat r_norm;

12linkfloat g_norm;

13linkfloat b_norm;

14linkfloat r_prim;

15linkfloat g_prim;

16linkfloat b_prim;

17link// interpolated texcoord (same name as in vertex shader)

18linkvarying vec2 vTexCoord;

19linkvoid main() {

20link  grayTextureColor = texture2D(texture, vTexCoord);

21link  red = grayTextureColor.r;

22link  r_norm = red/255.0;

23link  g_norm = gr/255.0;

24link  b_norm = blue/255.0;

25link  r_prim = 255.0 * pow(r_norm,0.45);

26link  g_prim = 255.0 * pow(g_norm,0.45);

27link  b_prim = 255.0 * pow(b_norm,0.45);

28link  gray = (r_prim*0.299 + g_prim*0.587 + b_prim*0.0114);

29link  grayTextureColor.r = gray;

30link  grayTextureColor.g = gray;

31link  grayTextureColor.b = gray;

32link  grayTextureColor.a = 1.0;

33link  gl_FragColor = grayTextureColor * vVertexColor;  

34link}

linkVideoLuma

El reto aquí es poder enviar un video a la textura ya que está solo recibe imagenes, la solución la encontramos con el parametro setUniform del shader pasandole como parametro, la textura y el video, y el alto y el ancho de la misma como lo veremos en la presentación del código.

1linklet theShader;

2linklet video;

3linklet planeSide = 400;

4linkfunction preload(){

5link  // load the shader

6link  theShader = loadShader('/vc/docs/sketches/LumaShader/Luma_Ponderado/texture.vert','/vc/docs/sketches/LumaShader/Luma_Ponderado/texture.frag');

7link  video = createVideo("/vc/docs/sketches/LumaShader/Luma_Video/SpaceJam.mp4");

8link}

9linkfunction setup() {

10link  // shaders require WEBGL mode to work

11link  createCanvas(400, 400, WEBGL);

12link  video.loop();

13link  video.hide();

14link  video.volume(0);

15link  noStroke();

16link  shader(theShader)

17link  textureMode(NORMAL);

18link  shader(theShader);

19link  // here we're using setUniform() to send our uniform values to the shader

20link}

21linkfunction draw() {

22link  background(255);

23link  beginShape();

24link  vertex(-planeSide/2, -planeSide/2, 0, 0); // esquina inferior izquierda

25link  vertex(planeSide/2, -planeSide/2, 1, 0); // esquina inferior derecha

26link  vertex(planeSide/2, planeSide/2, 1, 1); // esquina superior derecha

27link  vertex(-planeSide/2, planeSide/2, 0, 1); // esquina superior izquierda

28link  endShape(CLOSE);

29link  theShader.setUniform("texture", video);

30link  theShader.setUniform("textureWidth", 400.0);

31link  theShader.setUniform("textureHeight", 400.0)

32link}
