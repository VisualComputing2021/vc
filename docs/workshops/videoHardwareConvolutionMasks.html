linkConvolución en videos

Para los videos también se puede usar el poder de paralelización que ofrecen los núcleos de una gpu.

Se implementaron máscaras de convolución de 3x3, al igual que en las imágenes y se usa el mismo algoritmo, en el fragment shader se accede a los valores de color de los 8 texeles adyacentes de la imagen (sumado al texel actual), obteniendo 9 valores diferentes, los cuales se operan en una suma ponderada, donde los 9 pesos van indicados en el vector u_mask; y finalmente el valor resultante es asignado al fragmento correspondiente.

2linkprecision mediump float;

3link//uniform variables coming from p5

4linkuniform sampler2D u_texture;

5linkuniform vec2 u_img_unit;

6linkuniform float u_mask[9];

7link// interpolated color (same name as in vertex shader)

8linkvarying vec4 vVertexColor;

9link// interpolated texcoord (same name as in vertex shader)

10linkvarying vec2 vTexCoord;

11linkvoid main() {

12linkvec2 textureIndexes[9];

13linkint index = 0;

14linktextureIndexes[0] = vTexCoord + vec2(-u_img_unit.s, +u_img_unit.t);

15linktextureIndexes[1] = vTexCoord + vec2(          0.0, +u_img_unit.t);

16linktextureIndexes[2] = vTexCoord + vec2(+u_img_unit.s, +u_img_unit.t);

17linktextureIndexes[3] = vTexCoord + vec2(-u_img_unit.s,           0.0);

18linktextureIndexes[4] = vTexCoord + vec2(          0.0,           0.0);

19linktextureIndexes[5] = vTexCoord + vec2(+u_img_unit.s,           0.0);

20linktextureIndexes[6] = vTexCoord + vec2(-u_img_unit.s, -u_img_unit.t);

21linktextureIndexes[7] = vTexCoord + vec2(          0.0, -u_img_unit.t);

22linktextureIndexes[8] = vTexCoord + vec2(+u_img_unit.t, -u_img_unit.t);

23linkvec4 textureValues[9];

24linkfor(int i = 0; i < 9; i++){

25linktextureValues[i] = texture2D(u_texture, textureIndexes[i]);

26link}

27linkvec4 convolution = vec4(0, 0, 0, 0);

28linkfor(int i = 0; i < 9; i++){

29linkconvolution += textureValues[i] * u_mask[i];

30link}

31linkconvolution.a = 1.0;

32linkgl_FragColor = convolution * vVertexColor;  

33link}
